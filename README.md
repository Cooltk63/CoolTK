SELECT  
    REPLACE(REPLACE(SUBSTR(ch.HEDGEINV_INVDATE, 1, 10), '-', '-'), '-19', '-2019') AS datedto,  
    ch.HEDGEINV_PAN,  
    ch.HEDGEINV_CUSTOMER,  
    ch.HEDGEINV_AMOUNT, 
    ch.HEDGEINV_ID AS RWOSEQ, 
    ch.HEDGEINV_ID, 
    'false' 
FROM CRS_HEDGEINV ch  
WHERE ch.REPORT_MASTER_LIST_ID_FK = '6864028'
ORDER BY ch.HEDGEINV_ID;
